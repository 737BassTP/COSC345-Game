\begin{DoxyParams}{Parameters}
{\em from} & Start point. \\
\hline
{\em to} & End point. \\
\hline
{\em percentage} & Percentage between start-\/ and end point. \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
